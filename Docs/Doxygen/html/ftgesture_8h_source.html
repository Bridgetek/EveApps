306int16_t distbwPoints( int16_t x1,int16_t y1,int16_t x2,int16_t y2)
307{
308 int16_t dx,dy;
309 dx = x1-x2;
310 dy = y1-y2;
311 return sqrt(pow((double)dx,2)+pow((double)dy,2));
312}
325int16_t avgAngle(uint8_t tchs,int16_t cx,int16_t cy,int16_t currx[],\
326 int16_t curry[],int16_t prevx[],int16_t prevy[])
327{
328 int16_t i,a,b,angle = 0;
329 for(i=0;i<tchs;i++)
330 {
331 if(Isvalidtouch(currx[i]) && Isvalidtouch(prevx[i]))
332 {
333 a = anglebwPoints(cx,cy,currx[i],curry[i]);
334 b = anglebwPoints(cx,cy,prevx[i],prevy[i]);
335 angle += smallestAngleDiff(a,b);
336 }
337 }
338 return angle/=tchs;
339}
347int16_t smallestAngleDiff(int16_t a, int16_t b)
348{
349 int16_t d = a-b;
350 if(d>180)
351 d = d-360;
352 else if(d<-180)
353 d = d+360;
354 return d;
355}
365int16_t anglebwPoints(int16_t cx,int16_t cy,int16_t x,int16_t y)
366{
367 int16_t dx = x-cx;
368 int16_t dy = y-cy;
369 return RAD2DEGREE(atan2((double)dx,(double)dy));
370}
380int16_t avgScale(uint8_t tchs,int16_t ctr,int16_t curr[],int16_t prev[])
381{
382 int16_t distance = 0,i;
383 for(i=0;i<tchs;i++)
384 {
385 if(Isvalidtouch(curr[i]) && Isvalidtouch(prev[i]))
386 {
387 int16_t a = curr[i]-ctr;
388 int16_t b = prev[i]-ctr;
389 if(b<0) b = -b;
390 if(a<0) a = -a;
391 distance += (a-b);
392 }else
393 {
394 return 0;
395 }
396 }
397 return distance/=tchs;
398}

411 {
412 if(Isvalidtouch(prev[i]) && Isvalidtouch(curr[i]))
413 distance += (prev[i]-curr[i]);
414 }
415 return distance/=tchs;

425int16_t FTGLScrollerRun(uint8_t tchs,int16_t centerx,int16_t centery)
426{
427 int16_t change;
428 if(tchs)
429 {
430 if(Isvalidtouch(centerx))
431 {
432 if(FTGesture.scroller.mode&FTGL_SCROLLER_HORIZONTAL)
433 {
434 if(FTGesture.scrollerprev_centerx!=0)
435 FTGesture.scroller.velocityX = (FTGesture.scrollerprev_centerx - centerx) << FTGL_SUBPIXEL_PRESCISON;
436 }
437 }
438 if(Isvalidtouch(centery))
439 {
440 if(FTGesture.scroller.mode&FTGL_SCROLLER_VERTICAL)
441 {
442 if(FTGesture.scrollerprev_centery!=0)
443 FTGesture.scroller.velocityY = (FTGesture.scrollerprev_centery - centery) << FTGL_SUBPIXEL_PRESCISON;
444 }
445 }
446 if(Isvalidtouch(centerx))
447 FTGesture.scrollerprev_centerx = centerx;
448 if(Isvalidtouch(centery))
449 FTGesture.scrollerprev_centery = centery;
450 }
451 else
452 {
453 FTGesture.scrollerprev_centerx = 0;
454 FTGesture.scrollerprev_centery = 0;
455 if(FTGesture.scroller.mode&FTGL_SCROLLER_HORIZONTAL)
456 {
457 change = max(1, abs(FTGesture.scroller.velocityX) >> FTGL_SCROLLER_RATE);
458 if(FTGesture.scroller.velocityX<0)
459 FTGesture.scroller.velocityX+=change;
460 if(FTGesture.scroller.velocityX>0)
461 FTGesture.scroller.velocityX-=change;
462 }
463 if(FTGesture.scroller.mode&FTGL_SCROLLER_VERTICAL)
464 {
465 change = max(1, abs(FTGesture.scroller.velocityY) >> FTGL_SCROLLER_RATE);
466 if(FTGesture.scroller.velocityY<0)
467 FTGesture.scroller.velocityY+=change;
468 if(FTGesture.scroller.velocityY>0)
469 FTGesture.scroller.velocityY-=change;
470 }
471 }
472 return 0;
473}

524FTGLENUM_TTAP detectTap()
525{
526 int16_t distx;
527 if(FTGesture.currNumTchs)
528 {
529 ax = FTGesture.centerX;
530 ay = FTGesture.centerY;
531 if(FTGesture.penDowncts < FTGL_MINTOUCHPENDOWNCOUNT/2) // Ignore if number of touch change
532 {

534 FTGesture.tap_cx = FTGesture.centerX;
535 FTGesture.tap_cy = FTGesture.centerY;
536 return FTGL_TTAP_NONE;
537 }

539 if(FTGesture.penDowncts > FTGL_MAXTOUCHPENDOWNCOUNT &&\
540 FTGesture.penDowncts < FTGL_MAXLONGDURATIONCOUNT)
541 {
542 FTGesture.tap_cts = -1;
543 if(distx < FTGL_MAXPXLSDIRECTION_RANGE)
544 {
545 FTGesture.tapType.numberofTouches = (FTGLENUM_TPOINTS)FTGesture.tap_tchs;
546 return FTGL_TTAP_LONGTAP;
547 }
548 }
549 }else if(FTGesture.tap_tchs!=0)
550 {
551 if(FTGesture.penDowncts > 0 && FTGesture.penDowncts <= FTGL_MAXTOUCHPENDOWNCOUNT)
552 {
553 distx = 0;//distbwPoints(FTGesture.tap_cx,FTGesture.tap_cy,FTGesture.centerX,FTGesture.centerY);
554 if(distx < FTGL_MAXPXLSDIRECTION_RANGE)
555 FTGesture.tap_cts += 1;
556 else
557 {
558 FTGesture.tap_tchs = 0;
559 FTGesture.tap_cts = 0;
560 return FTGL_TTAP_TAPIGNORE;
561 }
562 }else if(FTGesture.penUpcts>FTGL_MAXTOUCHPENUPCOUNT)
563 {
564 if(FTGesture.tap_cts>0 && FTGesture.tap_cts <= FTGL_MAXTAPS)
565 {
566 FTGLENUM_TTAP temp_cts = (FTGLENUM_TTAP)FTGesture.tap_cts;
567 FTGesture.tapType.numberofTouches = (FTGLENUM_TPOINTS)FTGesture.tap_tchs;
568 FTGesture.tap_tchs = 0;
569 FTGesture.tap_cts = 0;
570 return temp_cts;
571 }
572 FTGesture.tap_cts = 0;
573 }
574 }
575 return FTGL_TTAP_NONE;

583{
584 int16_t dx,dy,sx,sy;
585 int16_t cx1,cy1;
586 if(FTGesture.currNumTchs)
587 {
588 if(!FTGesture.swipe_tchs || FTGesture.penDowncts < 2)
589 {
590 FTGesture.swipe_tchs = FTGesture.startNumTchs;
591 FTGesture.swipe_cx = FTGesture.centerX;
592 FTGesture.swipe_cy = FTGesture.centerY;
593
594 }
595 }else if(FTGesture.swipe_tchs)
596 {
597 FTGesture.swipeType.numberofTouches = (FTGLENUM_TPOINTS)FTGesture.swipe_tchs;
598 FTGesture.swipe_tchs = 0;
599 if(FTGesture.penDowncts > FTGL_MAXTOUCHPENDOWNCOUNT)
600 return FTGL_TSWIPE_SWIPEIGNORE;
601 if(FTGesture.penDowncts < 2)
602 return FTGL_TSWIPE_SWIPEIGNORE;
603 cx1 = FTGesture.centerX;
604 cy1 = FTGesture.centerY;
605 dx = sx = FTGesture.swipe_cx-cx1;
606 dy = sy = FTGesture.swipe_cy-cy1;
607 FTGesture.swipe_cy = 0;
608 FTGesture.swipe_cx = 0;
609 if(sx < 0) dx = -sx;
610 if(sy < 0) dy = -sy;
611 if(dx > FTGL_SWIPE_MIN_PXLS && dx < FTGL_SWIPE_MAX_PXLS)
612 {
613 if(sx>0)
614 return FTGL_TSWIPE_SWIPELEFT;
615 else if(sx<0)
616 return FTGL_TSWIPE_SWIPERIGHT;
617 }else if(dy > FTGL_SWIPE_MIN_PXLS && dy < FTGL_SWIPE_MAX_PXLS)
618 {
619 if(sy>0)
620 return FTGL_TSWIPE_SWIPEUP;
621 else if(sy<0)
622 return FTGL_TSWIPE_SWIPEDOWN;
623 }
624 }
625 return FTGL_TSWIPE_NONE;

634uint32_t FTGLRun(int16_t x[],int16_t y[])
635{
636 int16_t i;
637 uint32_t gstatus = 0;
638 int8_t transform = 0;
639 FTGLENUM_TTAP tap=FTGL_TTAP_NONE;
640 FTGLENUM_TSWIPE swipe=FTGL_TSWIPE_NONE;
641 updatePrevstate();
642 updateCurrstate(x,y);
643 if(FTGesture.prevNumTchs==0 && FTGesture.currNumTchs!=0)
644 updateStartstate();
645
646 if(FTGesture.currNumTchs)
647 {
648 if(FTGesture.currNumTchs==FTGesture.startNumTchs)
649 {
650 for(i=0;i<FTGesture.currNumTchs;i++)
651 {
652 if(Isvalidtouch(FTGesture.currX[i]) && Isvalidtouch(FTGesture.startX[i]))
653 {
654 if((FTGesture.currX[i]-FTGesture.startX[i])>0)
655 transform+=1;
656 else if((FTGesture.currX[i]-FTGesture.startX[i])<0)
657 transform-=1;
658 }
659 }
660 }else
661 {
662 updateStartstate();
663 transform = -1;
664 }
665 }
666
667 if(FTGesture.currNumTchs)
668 {
669 if(1)
670 {
671 i = avgCenter(FTGesture.currNumTchs,FTGesture.currX);
672 if(i)FTGesture.centerX = i;
673 i = avgCenter(FTGesture.currNumTchs,FTGesture.currY);
674 if(i)FTGesture.centerY = i;
675 FTGesture.transform.dragX += avgDistance(FTGesture.currNumTchs,FTGesture.currX,FTGesture.prevX);
676 FTGesture.transform.dragX = max(FTGesture.transform.minDragX, min(FTGesture.transform.dragX, FTGesture.transform.maxDragX));
677 FTGesture.transform.dragY += avgDistance(FTGesture.currNumTchs,FTGesture.currY,FTGesture.prevY);
678 FTGesture.transform.dragY = max(FTGesture.transform.minDragY, min(FTGesture.transform.dragY, FTGesture.transform.maxDragY));
679 if(FTGesture.currNumTchs>1 && transform==0)
680 {
681 FTGesture.transform.angleofRotation += avgAngle(FTGesture.currNumTchs,FTGesture.centerX,FTGesture.centerY,\
682 FTGesture.currX,FTGesture.currY,\
683 FTGesture.prevX,FTGesture.prevY);
684 if(FTGesture.transform.angleofRotation > 360)
685 FTGesture.transform.angleofRotation-=360;
686 if(FTGesture.transform.angleofRotation < -360)
687 FTGesture.transform.angleofRotation+=360;
688 if((FTGesture.transform.angleofRotation+10)!=FTGesture.prev_angle)
689 {
690 i = avgScale(FTGesture.currNumTchs,FTGesture.centerX,FTGesture.currX,FTGesture.prevX);
691 FTGesture.transform.scaleX+= i;
692 i = avgScale(FTGesture.currNumTchs,FTGesture.centerY,FTGesture.currY,FTGesture.prevY);
693 FTGesture.transform.scaleY+= i;
694 if(FTGesture.transform.scaleX>FTGesture.transform.maxScaleX) FTGesture.transform.scaleX = FTGesture.transform.maxScaleX;
