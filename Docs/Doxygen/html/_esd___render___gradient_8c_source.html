16ESD_CORE_EXPORT void Esd_Render_MultiGradient(int16_t x, int16_t y, int16_t width, int16_t height, esd_argb32_t topLeft, esd_argb32_t topRight, esd_argb32_t bottomLeft, esd_argb32_t bottomRight)
17{
18 EVE_HalContext *phost = Esd_GetHost();
19 Esd_Context *ec = Esd_CurrentContext;
20 uint32_t addr;
21 bool alpha;
22 union
23 {
24 uint16_t values[4];
25 uint64_t cache;
26 } color;
27
28 // Don't render empty
29 if (width == 0 || height == 0)
30 return;
31
32 // Get address of RAM_G used for gradient palette
33 addr = Esd_GpuAlloc_Get(Esd_GAlloc, ec->MultiGradientGpuHandle);
34 if (addr == GA_INVALID)
35 {
36 // Allocate enough memory for 32 gradients.
37 // Two bytes * four pixels * 64 gradients, 32 per frame flip.
38 ec->MultiGradientGpuHandle = Esd_GpuAlloc_Alloc(Esd_GAlloc, 2 * 4 * ESD_MULTIGRADIENT_MAX_NB, GA_GC_FLAG);
39 addr = Esd_GpuAlloc_Get(Esd_GAlloc, ec->MultiGradientGpuHandle);
40#ifdef ESD_MULTIGRADIENT_CACHE
41 memset(ec->MultiGradientCache, 0, sizeof(ec->MultiGradientCache));
42 EVE_CoCmd_memZero(phost, addr + (ec->MultiGradientCell * 8), 8);
43 ++ec->MultiGradientCell;
44 ec->MultiGradientCell &= (ESD_MULTIGRADIENT_MAX_NB - 1);
45#endif
46 }
47 if (addr == GA_INVALID)
48 {
49 eve_printf_debug("Unable to allocate RAM_G for MultiGradient\n");
50 return;
51 }
52
53 // Check if the colors have alpha, if so we'll use ARGB4, otherwise RGB565
54 alpha = topLeft < 0xFF000000 || topRight < 0xFF000000 || bottomLeft < 0xFF000000 || bottomRight < 0xFF000000;
55 if (alpha)
56 {
57 color.values[0] = ESD_COLOR_ARGB4(topLeft);
58 color.values[1] = ESD_COLOR_ARGB4(topRight);
59 color.values[2] = ESD_COLOR_ARGB4(bottomLeft);
60 color.values[3] = ESD_COLOR_ARGB4(bottomRight);
61 }
62 else
63 {
64 color.values[0] = ESD_COLOR_RGB565(topLeft);
65 color.values[1] = ESD_COLOR_RGB565(topRight);
66 color.values[2] = ESD_COLOR_RGB565(bottomLeft);
67 color.values[3] = ESD_COLOR_RGB565(bottomRight);
68 }
69
70 // Select cell address directly
71 addr += (ec->MultiGradientCell * 8);
72
73#ifdef ESD_MULTIGRADIENT_CACHE
74 bool cached = false;
75 for (int16_t i = 7; i >= 0; --i)
76 {
77 if (ec->MultiGradientCache[i] == color.cache)
78 {
79 // If i == (ec->MultiGradientCacheIdx - 1),
80 // then the cell will be (ec->MultiGradientCell - 1).
81 // Adjust addr accordingly.
82 int16_t cellOffset = ec->MultiGradientCacheIdx - i;
83 if (cellOffset <= 0)
84 cellOffset += 8;
85 addr -= ((uint16_t)cellOffset * 8);
86 cached = true;
87 break;
88 }
89 }
90 if (!cached)
91#endif
92 {
93 // Write gradient palette to RAM_G
94 EVE_CoCmd_memWrite(phost, addr, 8);
95 EVE_Cmd_wr16(phost, color.values[0]);
96 EVE_Cmd_wr16(phost, color.values[1]);
97 EVE_Cmd_wr16(phost, color.values[2]);
98 EVE_Cmd_wr16(phost, color.values[3]);
99#ifdef ESD_MULTIGRADIENT_CACHE
100 ec->MultiGradientCache[ec->MultiGradientCacheIdx] = color.cache;
101#endif
102
103 // Move to the next cell in the bitmap for next gradient
104 ++ec->MultiGradientCell;
105 ec->MultiGradientCell &= (ESD_MULTIGRADIENT_MAX_NB - 1);
106#ifdef ESD_MULTIGRADIENT_CACHE
107 ++ec->MultiGradientCacheIdx;
108 ec->MultiGradientCacheIdx &= 7;
109#endif
110 }
111
112 // Set required state
113 EVE_CoDl_colorArgb_ex(phost, ESD_ARGB_WHITE);
114
115 // Use the scratch handle
116 EVE_CoDl_bitmapHandle(phost, ESD_CO_SCRATCH_HANDLE);
117 EVE_CoDl_vertexFormat(phost, 0);
118 EVE_CoDl_begin(phost, BITMAPS);
119
120 /* ---- */
121 /* NOTE: Bypassing CoDl optimizer on purpose inside a SAVE_CONTEXT block */
122 /* ---- */
123
124 // Use local rendering context, bypass ESD display list functions.
125 // This is useful here, since we're changing bitmap transform matrices, which may use a lot of display list entries.
126 EVE_CoCmd_dl(Esd_Host, SAVE_CONTEXT());
127
128 // Initialize the bitmap options
129 if (EVE_CHIPID >= EVE_FT810)
130 {
131 EVE_CoCmd_dl(Esd_Host, BITMAP_LAYOUT_H(0, 0));
132 EVE_CoCmd_dl(Esd_Host, BITMAP_SIZE_H(width >> 9, height >> 9));
133 }
134 EVE_CoCmd_dl(Esd_Host, BITMAP_LAYOUT(alpha ? ARGB4 : RGB565, 4, 2));
135 EVE_CoCmd_dl(Esd_Host, BITMAP_SIZE(BILINEAR, REPEAT, REPEAT, width, height));
136
137 EVE_CoCmd_dl(Esd_Host, BITMAP_SOURCE(addr));
138
139 // Set the scaling matrix
140#if ESD_MULTIGRADIENT_CO_SCALE
141 EVE_CoCmd_loadIdentity(Esd_Host);
142 EVE_CoCmd_scale(Esd_Host, (int32_t)width << 16, (int32_t)height << 16);
143 EVE_CoCmd_setMatrix(Esd_Host);
144#else
145 if (EVE_CHIPID >= EVE_BT815)
146 {
147 EVE_CoCmd_dl(Esd_Host, BITMAP_TRANSFORM_A_EXT(1, 0x8000 / width));
148 EVE_CoCmd_dl(Esd_Host, BITMAP_TRANSFORM_E_EXT(1, 0x8000 / height));
149 }
150 else
151 {
152 EVE_CoCmd_dl(Esd_Host, BITMAP_TRANSFORM_A(0x0100 / width));
153 EVE_CoCmd_dl(Esd_Host, BITMAP_TRANSFORM_E(0x0100 / height));
154 }
155#endif
156
157 if (EVE_CHIPID >= EVE_BT815)
158 {
159 EVE_CoCmd_dl(Esd_Host, VERTEX2F(x, y));
160 }
161 else
162 {
163 EVE_CoCmd_dl(Esd_Host, VERTEX2II(x, y, ESD_CO_SCRATCH_HANDLE, 0));
164 }
165
166 // Restore rendering context, ESD display list optimizations functions should be used again after this.
167#if ESD_MULTIGRADIENT_CO_SCALE
168 EVE_CoCmd_loadIdentity(Esd_Host);
169#endif
170 EVE_CoCmd_dl(Esd_Host, RESTORE_CONTEXT());
171
172 /* ---- */
173 /* NOTE: Bypassed CoDl optimizer on purpose inside a SAVE_CONTEXT block */
174 /* ---- */
175
176 EVE_CoDl_end(phost);
177}
