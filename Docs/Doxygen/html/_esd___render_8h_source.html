28ESD_FUNCTION(ESD_COMPOSE_RGB_ALPHA, Type = esd_argb32_t, DisplayName = "Color A+RGB Combine", Category = EsdUtilities, Macro, Include = "Esd_Core.h")
29ESD_PARAMETER(rgb, Type = esd_rgb32_t, Default = #FFFFFF)
30ESD_PARAMETER(a, Type = uint8_t, Default = 255)
31#define ESD_COMPOSE_RGB_ALPHA(rgb, a) (((esd_argb32_t)rgb & 0xFFFFFF) | (((esd_argb32_t)a) << 24))
32
33#define ESD_DECOMPOSE_ALPHA(argb) ((argb) >> 24)
34#define ESD_DECOMPOSE_RED(rgb) (((rgb) >> 16) & 0xFF)
35#define ESD_DECOMPOSE_GREEN(rgb) (((rgb) >> 8) & 0xFF)
36#define ESD_DECOMPOSE_BLUE(rgb) ((rgb)&0xFF)
37
38#define ESD_COMPOSE_ARGB4(r, g, b, a) (((a & 0xF0) << 8) | ((r & 0xF0) << 4) | (g & 0xF0) | (b >> 4))
39#define ESD_COMPOSE_RGB565(r, g, b) (((r & 0xF8) << 8) | ((g & 0xFC) << 3) | (b >> 3))
40#define ESD_COMPOSE_ARGB1555(r, g, b, a) ((((a)&0x80) << 8) | (((r)&0xF8) << 7) | (((g)&0xF8) << 2) | ((b) >> 3))
41#define ESD_COMPOSE_ARGB8888(r, g, b, a) ((a) << 24) | ((r) << 16) | ((g) << 8) | (b)

43#define ESD_COLOR_ARGB4(argb) ESD_COMPOSE_ARGB4(ESD_DECOMPOSE_RED(argb), ESD_DECOMPOSE_GREEN(argb), ESD_DECOMPOSE_BLUE(argb), ESD_DECOMPOSE_ALPHA(argb))
