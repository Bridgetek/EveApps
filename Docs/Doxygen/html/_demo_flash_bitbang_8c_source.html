272 kb.type = ip->keyboard_type;
273 kb.w = KEY_W;
274 kb.h = KEY_W;
275 kb.font = 21;
276 kb.opt = 0;
277 kb.x = ip->x;

280 if (ip->isKeyboardActive == 0 &&
281 Gesture_Get()->tagReleased == ip->tag &&
282 isKbActive == 0) {
283 ip->isKeyboardActive = 1;
284 return;
285 }
286
287 if (ip->isKeyboardActive == 1) {
288 uint8_t key = keyboard_event(&kb);
289 if (key == KEYCODE_CLOSE) {
290 ip->isKeyboardActive = 0;
291 isKbActive = 0;
292 return;
293 }
294 else if (key == KEYCODE_BACK) {
295 if (ip->crrLen > 0) {
296 ip->currText[ip->crrLen - 1] = 0;
297 ip->crrLen--;
298 }
299 }
300 else if (key >= KEYTAG_MIN && key <= KEYTAG_MAX) {
301 if (INPUT_MAXLEN > ip->crrLen) {
302 ip->currText[ip->crrLen] = key;
303 ip->crrLen++;
304 }
305 }
306
307 keyboard_draw(kb);
308 }

395uint32_t sendtx(uint32_t num, uint8_t* data) {
396 const allign = 4;
397 int numW = ALIGN(num, allign);
398 int padding = numW - num;
399 isPreviousError = 0;
400
402 EVE_Cmd_wr32(s_pHalContext, CMD_FLASHSPIDESEL);
403 EVE_Cmd_waitFlush(s_pHalContext);
404
405 EVE_Cmd_wr32(s_pHalContext, CMD_FLASHSPITX);
406 EVE_Cmd_wr32(s_pHalContext, num);
407 for (int i = 0; i < num; i += allign) {
408 uint32_t d32 = 0;
409 uint8_t d8[4];
410 int k = 0;
411 for (int j = (allign - 1); j >= 0; j--) {
412 if ((i + j) < num) {
413 d8[k++] = data[i + j];
414 }
415 else {
416 d8[k++] = 0;
417 }
418 }
419 d32 = d8[0];
420 d32 = (d32 << 8) | d8[1];
421 d32 = (d32 << 8) | d8[2];
422 d32 = (d32 << 8) | d8[3];
423
424 EVE_Cmd_wr32(s_pHalContext, d32);
425 d32 = 0;
426 }
427 EVE_CoCmd_nop(s_pHalContext);
428 EVE_Cmd_waitFlush(s_pHalContext);
429 return;
430}

450uint8_t getParamtx(uint8_t* string, PARAM* pm) {
451 // command "89ABCDEF 89ABCDEF" = "89ABCDEF 89 AB CD EF"
452 // command " F 89ABCDEF" = " F 89 AB CD EF"
453 // command " F EF" = " F EF 00 00 00"
454 // command " F EF0" = " F EF 00 00 00"
455
456 uint8_t isParsingOk = 1;
457
458 pm->len32bit = 0;
459
460#define CHAR_PER_INT32 8 // 1 number 32 bit has 8 character 0->F
461 uint8_t param[CHAR_PER_INT32];
462 uint8_t index = 0;
463
464 for (int i = 0; string[i] != 0; i++)
465 {
466 int len = 0;
467 if (string[i] == ' ' || string[i + 1] == 0) {
468 if (string[i + 1] == 0) {
469 if (index >= CHAR_PER_INT32) {
470 isParsingOk = 0;
471 return isParsingOk;
472 }
473 param[index] = string[i];
474 index++;
475 }
476 if (pm->len32bit > 0) {
478 pm->len8bit = pm->len32bit * 4;
479 pm->arr32bit[pm->len32bit] = 0;
480
481 for (int j = 0; j < index; j += 2) {
482 uint8_t c[3];
483 c[0] = param[j + 0];
484 c[1] = param[j + 1];
485 c[2] = 0; // end of string
486 uint32_t num8bit = str2dec_32bit(c);
487
488 pm->arr8bit[pm->len8bit++] = num8bit;
489 }
490 pm->len32bit++;
491 }
492 else {
494 pm->arr32bit[pm->len32bit] = str2dec_32bit(param);
495 pm->len32bit++;
496 }
497
499 index = 0;
500 memset(param, 0, sizeof(param));
501 }
502 else {
503 if (index >= CHAR_PER_INT32) {
504 isParsingOk = 0;
505 return isParsingOk;
506 }
507
509 param[index] = string[i];
510 index++;
511 }
512 }
513
514 return isParsingOk;
515}
