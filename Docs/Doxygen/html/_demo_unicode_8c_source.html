267static int processEvents() {
268 int x = 600;
269 int y = 160;
270 int w = 40;
271 int typingLength = strlen(txtTyping);
272 static int counter = 0;
273 const int timeval = 10;
274
275 if (Gesture_Get()->tagReleased == buttonsCn.tag) {
276 crrTxtStr = txtStrCn;
277 crrFont = fontCn;
278 memset(txtTyping, 0, sizeof(txtTyping));
279 }
280 else if (Gesture_Get()->tagReleased == buttonsEn.tag) {
281 crrTxtStr = txtStrEn;
282 crrFont = fontEn;
283 memset(txtTyping, 0, sizeof(txtTyping));
284 }
285 else if (Gesture_Get()->tagReleased == buttonsJa.tag) {
286 crrTxtStr = txtStrJa;
287 crrFont = fontJa;
288 memset(txtTyping, 0, sizeof(txtTyping));
289 }
290 else if (Gesture_Get()->tagReleased == buttonsHi.tag) {
291 crrTxtStr = txtStrHi;
292 crrFont = fontHi;
293 memset(txtTyping, 0, sizeof(txtTyping));
294 }
295 else if (Gesture_Get()->tagReleased == TAG_TOGGLE) {
296 if (toggleState) {
297 toggleState = 0;
298 }
299 else {
300 toggleState = 65535;
301 }
302 }
303 else if (Gesture_Get()->tagReleased == TAG_KEY_BACKSPACE) {
304 if (typingLength > 0) {
305 txtTyping[typingLength - 1] = 0;
306 }
307 }
308 else if (Gesture_Get()->tagReleased > 0) {
309 if (typingLength > w) {
310 memcpy(txtTyping, &txtTyping[1], typingLength);
311 typingLength = w;
312 txtTyping[typingLength] = 0;
313 }
314
315 txtTyping[typingLength] = Gesture_Get()->tagReleased;
316 }
317 else {
318 }
319
320 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(200, 200, 200));
321 EVE_Cmd_wr32(s_pHalContext, LINE_WIDTH(PX(20)));
322 EVE_Cmd_wr32(s_pHalContext, BEGIN(RECTS));
323 EVE_Cmd_wr32(s_pHalContext, VERTEX2F(PX(70), PX(y - 10)));
324 EVE_Cmd_wr32(s_pHalContext, VERTEX2F(PX(10 + x), PX(y + 50)));
325 EVE_Cmd_wr32(s_pHalContext, END());
326
327 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(0, 0, 0));
328
329 if (txtTyping[0] != 0) {
330 EVE_CoCmd_text(s_pHalContext, x, y, crrFont, OPT_RIGHTX, txtTyping);
331 }
332
333 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(255, 255, 255));
334 EVE_Cmd_wr32(s_pHalContext, LINE_WIDTH(PX(20)));
335 EVE_Cmd_wr32(s_pHalContext, BEGIN(RECTS));
336 EVE_Cmd_wr32(s_pHalContext, VERTEX2F(PX(0), PX(y - 10)));
337 EVE_Cmd_wr32(s_pHalContext, VERTEX2F(PX(70 - 20*2), PX(y + 50)));
338 EVE_Cmd_wr32(s_pHalContext, END());
339
340 if (counter > timeval) {
341 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(0, 0, 255));
342 EVE_CoCmd_text(s_pHalContext, x, y, 29, 0, "|");
343 }
344
345 counter++;
346 counter = counter % (timeval * 2);
347
348 txtToggle = crrTxtStr[eTxtIdOff];
349 if (toggleState) {
350 txtToggle = crrTxtStr[eTxtIdOn];
351 }
352 return 0;
353}

363 uint8_t str[2];
