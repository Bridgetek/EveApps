489int16_t App_TrsMtrxRotate(App_GraphicsMatrix_t *pMatrix, int32_t th) {
490 double pi = 3.1415926535;
491 double angle = -th*pi / 180;
492 App_GraphicsMatrix_t tempmtx, tempinput;
493
494 tempinput.a = pMatrix->a;
495 tempinput.b = pMatrix->b;
496 tempinput.c = pMatrix->c;
497 tempinput.d = pMatrix->d;
498 tempinput.e = pMatrix->e;
499 tempinput.f = pMatrix->f;
500
501 App_TrsMtrxLoadIdentity(&tempmtx);
502 tempmtx.a = cos(angle)*APP_MATIRX_PRECITION;
503 tempmtx.b = sin(angle)*APP_MATIRX_PRECITION;
504 tempmtx.d = -sin(angle)*APP_MATIRX_PRECITION;
505 tempmtx.e = cos(angle)*APP_MATIRX_PRECITION;
506
507 /* perform matrix multiplecation and store in the input */
508 pMatrix->a = tempinput.a*tempmtx.a + tempinput.b*tempmtx.d;
509 pMatrix->b = tempinput.a*tempmtx.b + tempinput.b*tempmtx.e;

512 pMatrix->e = tempinput.d*tempmtx.b + tempinput.e*tempmtx.e;
513 pMatrix->f = tempinput.d*tempmtx.c + tempinput.e*tempmtx.f + tempinput.f * 1;
514
515 return 0;
516}
