88int32_t Esd_LittleFs_LoadFlashAddress(int32_t *addr, const char *file, uint8_t *metadata)
89{
90 int err;
91 lfs_file_t lfsFile;
92 uint8_t buf[EVE_FLASH_WRITE_ALIGN];
93 struct lfs_file_config fileConfig = { 0 };
94 if (!file)
95 return FA_INVALID; // No file name specified
96 if (*addr >= EVE_FLASH_FIRMWARE_SIZE)
97 return *addr; // Flash address already cached
98 if (*addr != FA_INVALID)
99 return FA_INVALID; // Flash file not found or not reserved
100 if (!Esd_LittleFS_Mount())
101 return FA_INVALID; // LittleFS cannot be mounted
102 // eve_printf("[DIAGNOSTIC] Load flash address %s\n", file);
103 fileConfig.buffer = buf;
104 if (err = lfs_file_opencfg(&Esd_CurrentContext->Lfs, &lfsFile, file, LFS_O_RDONLY, &fileConfig))
105 {
106 if (err == LFS_ERR_NOENT)
107 {
108 *addr = FA_NOTFOUND;
109 eve_printf_debug("Flash file doesn't exist '%s'\n", file);
110 }
111 else
112 *addr = FA_INVALID;
113 return FA_INVALID; // File doesn't exist
114 }
115 lfs_block_t head;
116 if (err = lfs_file_reserved(&Esd_CurrentContext->Lfs, &lfsFile, &head))
117 {
118 if (err == LFS_ERR_INVAL)
119 {
120 eve_printf_debug("Flash file is not reserved, load as a normal file '%s'\n", file);
121 *addr = FA_NOTRESERVED; // This is a regular file, cannot access directly
122 }
123 else
124 *addr = FA_INVALID;
125 }
126 else
127 {
128 *addr = EVE_FLASH_FIRMWARE_SIZE + (head * Esd_CurrentContext->Lfs.cfg->block_size);
129 }
130 lfs_file_close(&Esd_CurrentContext->Lfs, &lfsFile);
131 if (metadata)
132 Esd_LittleFs_ReadMetadataFile(metadata, file);
133 // if (*addr >= EVE_FLASH_FIRMWARE_SIZE)
134 // eve_printf("[DIAGNOSTIC] Got flash address %i\n", (int)(*addr));
135 return (*addr >= EVE_FLASH_FIRMWARE_SIZE) ? *addr : FA_INVALID; // Return address in flash
136}

144static bool Esd_LoadVideoFrameFromFile(uint32_t *imageFormat, uint32_t dst, const char *file)
145{
146 EVE_HalContext *phost = Esd_GetHost();
147 if (phost->CmdFault)
148 return false;
149
150 if (!EVE_Hal_supportVideo(phost))
151 {
152 eve_assert_ex(false, "Esd_LoadVideoFrameFromFile is not available on the current graphics platform\n");
153 return false;
154 }
155
156 if (!EVE_Util_sdCardReady(phost))
157 {
158 eve_printf_debug("SD card not ready\n");
159 return false;
160 }
161
162 /* Allocate RAM_G space for FIFO and completion pointer */
163 uint32_t fifoSize = 16 * 1024; /* TODO: What's an ideal FIFO size? */
164 Esd_GpuHandle fifoHandle = Esd_GpuAlloc_Alloc(Esd_GAlloc, fifoSize + 4, 0);
165 uint32_t fifoAddr = Esd_GpuAlloc_Get(Esd_GAlloc, fifoHandle);
166 if (fifoAddr == GA_INVALID)
167 {
168 return false;
169 }
170
171 /* Setup media FIFO */
172 if (!EVE_MediaFifo_set(phost, fifoAddr, fifoSize))
173 {
174 Esd_GpuAlloc_Free(Esd_GAlloc, fifoHandle);
175 return false;
176 }
177
178 uint16_t bkpCmdDl;
179 uint32_t bkpDl[10];
180
181 if (EVE_CHIPID < EVE_BT815)
182 {
183 if (EVE_Cmd_waitFlush(phost))
184 {
185 /* Save */
186 bkpCmdDl = EVE_Hal_rd16(phost, REG_CMD_DL);
187 if (bkpCmdDl)
188 {
189 for (int i = 0; i < 10; ++i)
190 bkpDl[i] = EVE_Hal_rd32(phost, RAM_DL + 4 * i);
191 }
192 }
193 else
194 {
195 bkpCmdDl = 0;
196 }
197 }
198
199 /* Load the first video frame */
200 uint32_t transfered = 0;
201 uint32_t ptr = fifoAddr + fifoSize;
202 EVE_CoCmd_videoStart(phost);
203 EVE_CoCmd_videoFrame(phost, dst, ptr);
204 bool res = EVE_Util_loadMediaFile(phost, file, &transfered);
205 EVE_Util_closeFile(phost);
206
207 if (EVE_CHIPID < EVE_BT815)
208 {
209 if (bkpCmdDl && EVE_Cmd_waitFlush(phost))
210 {
211 /* Restore */
212 EVE_Hal_wr16(phost, REG_CMD_DL, bkpCmdDl);
213 for (int i = 0; i < 10; ++i)
214 EVE_Hal_wr32(phost, RAM_DL + 4 * i, bkpDl[i]);
215 }
216 }
217
218 /* Release */
219 EVE_MediaFifo_close(phost);
220 Esd_GpuAlloc_Free(Esd_GAlloc, fifoHandle);
221 *imageFormat = RGB565;
222
223 if (EVE_Cmd_rp(phost) != EVE_Cmd_wp(phost))
224 {
225 eve_printf_debug("Coprocessor did not complete command, force fault\n");
226 EVE_Util_forceFault(phost, "ESD Core: CMD_VIDEOSTART and CMD_VIDEOFRAME aborted");
227 }
228
229 return res;
230}

267ESD_CORE_EXPORT bool Esd_LoadBitmapMetadata(Esd_BitmapInfo *bitmapInfo, uint8_t *metadata)
268{
269 if (metadata && metadata[ESD_METADATA_SIGNATURE])
270 {
271 uint32_t esdSignature = ESD_RD32_LE(metadata, ESD_METADATA_SIGNATURE);
272 if (esdSignature != ESD_RESOURCE_BMP)
273 {
274#ifdef ESD_BITMAPINFO_DEBUG
275 eve_printf_debug("Bitmap has an invalid metadata signature (0x%x)\n", (int)esdSignature);
276#endif
277 return false;
278 }
279
280 // Update compression and extracted size from metadata
281 uint8_t compression = metadata[ESD_METADATA_COMPRESSION];
282 ESD_METADATA_SET(bool, bitmapInfo, Compressed, compression == ESD_RESOURCE_DEFLATE, bitmapInfo->File);
283 ESD_METADATA_SET(int32_t, bitmapInfo, Size, ESD_RD32I_LE(metadata, ESD_METADATA_RAWSIZE), bitmapInfo->File);
284
285 // Update bitmap specific options from metadata
286 ESD_METADATA_SET(int32_t, bitmapInfo, Width, ESD_RD32I_LE(metadata, ESD_METADATA_WIDTH), bitmapInfo->File);
287 ESD_METADATA_SET(int32_t, bitmapInfo, Height, ESD_RD32I_LE(metadata, ESD_METADATA_HEIGHT), bitmapInfo->File);
288 ESD_METADATA_SET(int32_t, bitmapInfo, Stride, ESD_RD32I_LE(metadata, ESD_METADATA_STRIDE), bitmapInfo->File);
289 ESD_METADATA_SET_EX(uint32_t, bitmapInfo, Format, ESD_RD32_LE(metadata, ESD_METADATA_FORMAT), bitmapInfo->File, bitmapInfo->CoLoad = false; bitmapInfo->Video = false);
290 ESD_METADATA_SET(int16_t, bitmapInfo, Cells, ESD_RD16_LE(metadata, ESD_METADATA_CELLS), bitmapInfo->File);
291#if (EVE_SUPPORT_CHIPID >= EVE_BT815)
292 uint16_t swizzle = ESD_RD16_LE(metadata, ESD_METADATA_SWIZZLE);
293 bool hasSwizzle = swizzle & (1 << 12);
294 if (hasSwizzle)
295 {
296 ESD_METADATA_SET(uint16_t, bitmapInfo, SwizzleB, (swizzle >> 9) & 7, bitmapInfo->File);
297 ESD_METADATA_SET(uint16_t, bitmapInfo, SwizzleG, (swizzle >> 6) & 7, bitmapInfo->File);
298 ESD_METADATA_SET(uint16_t, bitmapInfo, SwizzleR, (swizzle >> 3) & 7, bitmapInfo->File);
299 ESD_METADATA_SET(uint16_t, bitmapInfo, SwizzleA, swizzle & 7, bitmapInfo->File);
300 }
301 ESD_METADATA_SET(bool, bitmapInfo, Swizzle, hasSwizzle, bitmapInfo->File);
302#endif
303
304#ifdef ESD_BITMAPINFO_DEBUG
305 // eve_printf_debug("Bitmap info metadata reloaded\n");
306#endif
307 }
308
309 // Okay if not loaded or if signature didn't fail
310 return true;
311}

324ESD_CORE_EXPORT uint32_t Esd_LoadBitmapEx(Esd_BitmapInfo *bitmapInfo, uint8_t *metadata)
325{
326 EVE_HalContext *phost = Esd_GetHost();
327 int32_t flashAddr = FA_INVALID;
328 uint32_t addr;
329 (void)phost;
330
331 if (metadata)
332 metadata[ESD_METADATA_SIGNATURE] = '\0';
333
334 if (!bitmapInfo)
335 {
336#ifdef ESD_BITMAPINFO_DEBUG
337 eve_printf_debug("Bitmap info is NULL\n");
338#endif
339 return GA_INVALID;
340 }
341
342 // If PreferRam is set, skip this and try to load to RAM_G first
343#ifdef EVE_FLASH_AVAILABLE
344 if (!bitmapInfo->PreferRam && ESD_BITMAPINFO_SUPPORT_DIRECT_FLASH(bitmapInfo))
345 {
346 // Just get the flash address if that's what we want
347
348 flashAddr = Esd_LittleFs_LoadFlashAddress(&bitmapInfo->FlashAddress, bitmapInfo->File, metadata);
349 if (!Esd_LoadBitmapMetadata(bitmapInfo, metadata))
350 return GA_INVALID;
351 // Ensure any new metadata still supports direct flash
352 if (!bitmapInfo->Compressed && ESD_IS_FORMAT_ASTC(bitmapInfo->Format))
353 {
354 if (flashAddr != FA_INVALID)
355 return ESD_DL_FLASH_ADDRESS(flashAddr);
356 return GA_INVALID;
357 }
358 }
359#else
360 if (bitmapInfo->Flash)
361 {
362#ifdef ESD_BITMAPINFO_DEBUG
363 eve_printf_debug("Flash storage not supported on this platform\n");
364#endif
365 return GA_INVALID;
366 }
367#endif
368
369 // Get address of specified handle
370 // eve_printf_debug("%i: %i\n", bitmapInfo->GpuHandle.Id, bitmapInfo->GpuHandle.Seq);
371 addr = Esd_GpuAlloc_Get(Esd_GAlloc, bitmapInfo->GpuHandle);
372 if (addr == GA_INVALID)
373 {
374 if (bitmapInfo->Flash)
375 {
376 flashAddr = Esd_LittleFs_LoadFlashAddress(&bitmapInfo->FlashAddress, bitmapInfo->File, metadata);
377 if (flashAddr == FA_INVALID)
378 {
379 /* TODO: This could be a regular LittleFS file, load to RAM_G similar to SD */
380 }
381 }
382 if (bitmapInfo->Flash ? (flashAddr == FA_INVALID) : !bitmapInfo->File)
383 {
384#ifdef ESD_BITMAPINFO_DEBUG
385 /*
386 if (bitmapInfo->Flash)
387 eve_printf_debug("Bitmap flash address is not found, cannot auto-load\n");
388 else
389 eve_printf_debug("Bitmap info file name is NULL, cannot auto-load\n");
390 */
391#endif
392 return GA_INVALID;
393 }
394
395 // Load metadata from file on SD card
396 if (metadata && !bitmapInfo->Flash && bitmapInfo->File)
397 {
398 // Generate metafile name
399 size_t nameLen = strlen(bitmapInfo->File);
400 if (nameLen + 6 > 128) // ".esdm" NUL
401 return GA_INVALID;
402 char metaFile[128];
403 strcpy_s(metaFile, sizeof(metaFile), bitmapInfo->File);
404 strcpy_s(&metaFile[nameLen], sizeof(metaFile) - nameLen, ".esdm");
405
406 // Try to load
407 EVE_Util_readFile(phost, metadata, ESD_METADATA_MAX, metaFile);
408 }
409
410#ifdef ESD_BITMAPINFO_DEBUG
411#ifndef ESD_LITTLEFS_FLASH
412 if (bitmapInfo->Flash)
413 eve_printf_debug("[DEVELOPER] Loading flash addr %i\n", bitmapInfo->FlashAddress);
414 else
415#endif
416 eve_printf_debug("[DEVELOPER] Loading '%s'\n", bitmapInfo->File);
417#endif
418
419 // Load bitmap metadata
420 Esd_LoadBitmapMetadata(bitmapInfo, metadata);
421
422 // Not loaded, load this bitmap
423 bitmapInfo->GpuHandle = Esd_GpuAlloc_Alloc(Esd_GAlloc, bitmapInfo->Size,
424 (bitmapInfo->Persistent ? 0 : GA_GC_FLAG) | ((bitmapInfo->Flash && bitmapInfo->PreferRam) ? GA_LOW_FLAG : 0));
425 addr = Esd_GpuAlloc_Get(Esd_GAlloc, bitmapInfo->GpuHandle);
426 if (addr != GA_INVALID)
427 {
428 bool coLoad = bitmapInfo->CoLoad || bitmapInfo->Format == JPEG || bitmapInfo->Format == PNG || bitmapInfo->Format == AVI;
429 bitmapInfo->CoLoad = coLoad;
430
431 bool video = bitmapInfo->Video || bitmapInfo->Format == AVI;
432 bitmapInfo->Video = video;
433
434 if (video && !EVE_Hal_supportVideo(phost))
435 {
436#ifdef ESD_BITMAPINFO_DEBUG
437 eve_printf_debug_once("Video is not supported\n");
438#endif
439 /* Video not supported */
440 /* TODO: We *could* do a software AVI parse, and submit it as JPG */
441 Esd_GpuAlloc_Free(Esd_GAlloc, bitmapInfo->GpuHandle);
442 return GA_INVALID;
443 }
444
445#ifdef ESD_BITMAPINFO_DEBUG
446 eve_printf_debug("Allocated space for bitmap\n");
447#endif
448
449 // Allocation space OK
450 if (
451#if defined(EVE_SUPPORT_VIDEO)
452 video ? (
453#ifdef EVE_FLASH_AVAILABLE
454 bitmapInfo->Flash ? !Esd_LoadVideoFrameFromFlash(&bitmapInfo->Format, addr, flashAddr) :
455#endif
456 !Esd_LoadVideoFrameFromFile(&bitmapInfo->Format, addr, bitmapInfo->File))
457 :
458#endif
459 (
460#ifdef EVE_FLASH_AVAILABLE
461 bitmapInfo->Flash ? !Esd_LoadFromFlash(coLoad ? &bitmapInfo->Format : NULL, bitmapInfo->Compressed, addr, flashAddr, bitmapInfo->Size) :
462#endif
463 !Esd_LoadFromFile(coLoad ? &bitmapInfo->Format : NULL, bitmapInfo->Compressed, addr, bitmapInfo->File)))
464 {
465#ifdef ESD_BITMAPINFO_DEBUG
466 eve_printf_debug(bitmapInfo->Flash ? "Failed to load bitmap from flash\n" : "Failed to load bitmap from file\n");
467#endif
468 // Failed to load from file
469 Esd_GpuAlloc_Free(Esd_GAlloc, bitmapInfo->GpuHandle);
470 addr = GA_INVALID;
471 }
472
473 if (coLoad && (bitmapInfo->Format == PALETTED565 || bitmapInfo->Format == PALETTED4444 || bitmapInfo->Format == L8))
474 {
475 // Truncate co loaded PNG that has paletted or L8 format
476 Esd_GpuAlloc_Truncate(Esd_GAlloc, bitmapInfo->GpuHandle, ((bitmapInfo->Format != L8) ? 512 : 0) + (bitmapInfo->Stride * bitmapInfo->Height));
477 }
478
479#ifdef ESD_COMPATIBILITY_ADDITIONALFILE
480 int32_t additionalFlashAddr = bitmapInfo->Flash ? Esd_LittleFs_LoadFlashAddress(&bitmapInfo->AdditionalFlashAddress, bitmapInfo->AdditionalFile, NULL) : FA_INVALID;
481 // Special load for DXT1, additional file loaded halfway the address space
482 if (bitmapInfo->Flash ? (additionalFlashAddr != FA_INVALID) : (!!bitmapInfo->AdditionalFile))
483 {
484 if (
485#ifdef EVE_FLASH_AVAILABLE
486 bitmapInfo->Flash ? !Esd_LoadFromFlash(coLoad ? &bitmapInfo->Format : NULL, bitmapInfo->Compressed, addr + (bitmapInfo->Size >> 1), additionalFlashAddr, bitmapInfo->Size) :
487#endif
488 !Esd_LoadFromFile(coLoad ? &bitmapInfo->Format : NULL, bitmapInfo->Compressed, addr + (bitmapInfo->Size >> 1), bitmapInfo->AdditionalFile))
489 {
490#ifdef ESD_BITMAPINFO_DEBUG
491 eve_printf_debug(bitmapInfo->Flash ? "Failed to load additional bitmap from flash\n" : "Failed to load additional bitmap from file\n");
492#endif
493 // Failed to load from additional file
494 // Esd_GpuAlloc_Free(Esd_GAlloc, bitmapInfo->GpuHandle);
495 addr = GA_INVALID;
496 }
497 }
498#endif
499 }
500 else
501 {
502#ifdef ESD_BITMAPINFO_DEBUG
503 eve_printf_debug("Unable to allocate space for bitmap\n");
504#endif
505 }
506 }
507
508#ifdef EVE_FLASH_AVAILABLE
509 if (addr == GA_INVALID && ESD_BITMAPINFO_SUPPORT_DIRECT_FLASH(bitmapInfo))
510 {
511 // Just get the flash address anyway if we failed to load (in case of PreferRam)
512 if (flashAddr != FA_INVALID)
513 return ESD_DL_FLASH_ADDRESS(flashAddr);
514 return GA_INVALID;
515 }
516#endif
517
518 if (addr != GA_INVALID)
519 {
520 addr = ESD_DL_RAM_G_ADDRESS(addr);
521 }
522
523 return addr;
524}

532ESD_CORE_EXPORT uint32_t Esd_LoadPalette(Esd_BitmapInfo *bitmapInfo)
533{
534 EVE_HalContext *phost = Esd_GetHost();
535 uint32_t addr;
536 (void)phost;
537
538 if (!bitmapInfo)
539 {
540#ifdef ESD_BITMAPINFO_DEBUG
541 eve_printf_debug("Bitmap info is NULL\n");
542#endif
543 return GA_INVALID;
544 }
545
546 if (!ESD_IS_FORMAT_PALETTED(bitmapInfo->Format))
547 {
548 // Not paletted
549 return GA_INVALID;
550 }
551
552#ifndef EVE_FLASH_AVAILABLE
553 if (bitmapInfo->Flash)
554 {
555#ifdef ESD_BITMAPINFO_DEBUG
556 // eve_printf_debug("Flash storage not supported on this platform\n");
557#endif
558 return GA_INVALID;
559 }
560#endif
561
562 if (EVE_CHIPID >= EVE_FT810)
563 {
564 // Don't load if no palette file
565#ifndef ESD_LITTLEFS_FLASH
566 if (bitmapInfo->Flash)
567 {
568 if (bitmapInfo->PaletteFlashAddress < EVE_FLASH_FIRMWARE_SIZE)
569 return GA_INVALID;
570 }
571 else
572#endif
573 if (!bitmapInfo->PaletteFile)
574 {
575 return GA_INVALID;
576 }
577
578 // Get palette address of specified handle
579 addr = Esd_GpuAlloc_Get(Esd_GAlloc, bitmapInfo->PaletteGpuHandle);
580 if (addr == GA_INVALID)
581 {
582 static const uint32_t sizes[PALETTED8 - PALETTED565 + 1] = { 256 * 2, 256 * 2, 256 * 4 };
583 uint32_t size = sizes[bitmapInfo->Format - PALETTED565];
584
585 // Verify if there is a file name or flash address specified, or if the flash address can be found in the flash filesystem
586 int32_t paletteFlashAddr = bitmapInfo->Flash ? Esd_LittleFs_LoadFlashAddress(&bitmapInfo->PaletteFlashAddress, bitmapInfo->PaletteFile, NULL) : FA_INVALID;
587 if (bitmapInfo->Flash ? (paletteFlashAddr == FA_INVALID) : (!bitmapInfo->PaletteFile))
588 {
589#ifdef ESD_BITMAPINFO_DEBUG
590 // eve_printf_debug("Bitmap info palette file name is NULL, cannot auto-load\n");
591#endif
592 return GA_INVALID;
593 }
594
595 // Not loaded, load this bitmap palette
596 bitmapInfo->PaletteGpuHandle = Esd_GpuAlloc_Alloc(Esd_GAlloc, size, bitmapInfo->Persistent ? 0 : GA_GC_FLAG);
597 addr = Esd_GpuAlloc_Get(Esd_GAlloc, bitmapInfo->PaletteGpuHandle);
598 if (addr != GA_INVALID)
599 {
600#ifdef ESD_BITMAPINFO_DEBUG
601 eve_printf_debug("Allocated space for bitmap palette\n");
602#endif
603 // Allocation space OK
604 if (
605#ifdef EVE_FLASH_AVAILABLE
606 bitmapInfo->Flash ? !EVE_CoCmd_flashRead_flush(Esd_Host, addr, paletteFlashAddr, size) :
607#endif
608 !EVE_Util_loadRawFile(Esd_Host, addr, bitmapInfo->PaletteFile))
609 {
610#ifdef ESD_BITMAPINFO_DEBUG
611 eve_printf_debug(bitmapInfo->Flash ? "Failed to load palette from flash\n" : "Failed to load palette from file\n");
612#endif
