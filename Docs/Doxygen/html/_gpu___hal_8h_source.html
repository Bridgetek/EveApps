498static int32_t Gpu_Hal_Dec2Ascii(char8_t *pSrc, int32_t value)
499{
500 int16_t Length;
501 char8_t *pdst;
502 char8_t charval;
503 int32_t CurrVal = value;
504 int32_t tmpval;
505 int32_t i;
506 char8_t tmparray[16];
507 char8_t idx = 0;
508
509 Length = (int16_t)strlen((char *)pSrc);
510 pdst = pSrc + Length;
511
512 if (0 == value)
513 {
514 *pdst++ = '0';
515 *pdst++ = '\0';
516 return 0;
517 }
518
519 if (CurrVal < 0)
520 {
521 *pdst++ = '-';
522 CurrVal = -CurrVal;
523 }
524 /* insert the value */
525 while (CurrVal > 0)
526 {
527 tmpval = CurrVal;
528 CurrVal /= 10;
529 tmpval = tmpval - CurrVal * 10;
530 charval = '0' + tmpval;
531 tmparray[idx++] = charval;
532 }
533
534 for (i = 0; i < idx; i++)
535 {
536 *pdst++ = tmparray[idx - i - 1];
537 }
538 *pdst++ = '\0';
539
540 return 0;
541}
