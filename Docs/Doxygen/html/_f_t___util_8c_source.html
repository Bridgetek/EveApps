290static void service_playback(void)
291{
292 UINT bytesread;
293
294 unsigned rp = Ft_Gpu_Hal_Rd32(g_GuiInternal.pdirect, REG_PLAYBACK_READPTR) - SCRATCH_AUDIO_BUF;
295 unsigned fullness = (g_GuiInternal.audio.wp - rp) & (PLAYBACK_BUFFER_SIZE - 1);
296
297 if (!test_bit(g_GuiInternal.audio.flags, FT_AUDIO_FLAGS_BIT_PLAYING)) {
298 return;
299 }
300 g_GuiInternal.audio.played_samples += (rp - g_GuiInternal.audio.prev_rdptr) & (PLAYBACK_BUFFER_SIZE - 1);
301 g_GuiInternal.audio.prev_rdptr = rp;
302
303 if (((ft_int32_t)(g_GuiInternal.audio.played_samples - g_GuiInternal.audio.last_sample) >= 0)
304 && (rp >= (unsigned) g_GuiInternal.audio.wp)) {
305 g_GuiInternal.audio.started_samples = g_GuiInternal.audio.last_sample;
306 g_GuiInternal.audio.last_sample += g_GuiInternal.audio.nsamples;
307 if ((g_GuiInternal.audio.mode == PLAY_ONCE || g_GuiInternal.audio.mode == PLAY_WAIT)) {
308 //disable_speaker(AUDIO_SOUND);
309 Ft_Gpu_Hal_Wr32(g_GuiInternal.pdirect, REG_PLAYBACK_FREQ, 100);
310 Gpu_CoCmd_MemSet(g_GuiInternal.pdirect, SCRATCH_AUDIO_BUF, 0x00000000,
311 PLAYBACK_BUFFER_SIZE);
312 g_GuiInternal.audio.mode = 255;
313 }
314 }
315
316 while (fullness < (PLAYBACK_BUFFER_SIZE - 1)) {
317 ft_uint8_t sample = 0;
318
319 if (test_bit(g_GuiInternal.audio.flags, FT_AUDIO_FLAGS_BIT_PLAYING)) {
320 if (g_GuiInternal.audio.adpcm) {
321 // FILE *f = fopen("dump", "w");
322 // while (++audio.atsample != audio.nsamples) {
323 // int16_t s = adpcm();
324 // fwrite(&s, 1, sizeof(s), f);
325 // }

327 // assert(0);
328 sample = adpcm() >> 8;
329 } else if (g_GuiInternal.audio.bps == 16) {
330 ft_int16_t s16[2];
331 if (g_GuiInternal.audio.channels == 1) {
332 //cmd_FRead(audio.f, (uint8_t*) &s16[0], sizeof(s16[0]), bytesread);
333 F_READ(g_GuiInternal.audio.file, (ft_uint8_t*)&s16[0], sizeof(s16[0]), bytesread);
334 sample = (s16[0] >> 8);
335 } else {
336 //cmd_FRead(audio.f, (uint8_t*) &s16, sizeof(s16), bytesread);
337 F_READ(g_GuiInternal.audio.file, (ft_uint8_t*) &s16, sizeof(s16), bytesread);
338 sample = (s16[0] + s16[1]) >> 9;
339 }
340 } else if ((g_GuiInternal.audio.format == LINEAR_SAMPLES) && (g_GuiInternal.audio.bps == 8)) {
341 ft_uint8_t u8[2];
342 if (g_GuiInternal.audio.channels == 1) {
343 //cmd_FRead(audio.f, (uint8_t*) &u8[0], sizeof(u8[0]), bytesread);
344 F_READ(g_GuiInternal.audio.file, (ft_uint8_t*) &u8[0], sizeof(u8[0]), bytesread);
345 sample = u8[0];
346 } else {
347 //cmd_FRead(audio.f, (uint8_t*) &u8, sizeof(u8), bytesread);
348 F_READ(g_GuiInternal.audio.file, (ft_uint8_t*) &u8, sizeof(u8), bytesread);
349 sample = (u8[0] + u8[1]) >> 1;
350 }
351 sample -= 128;
352 } else if (g_GuiInternal.audio.format == ULAW_SAMPLES) {
353 ft_uint8_t u8[2];
354 //cmd_FRead(audio.f, (uint8_t*) &u8, audio.channels, bytesread);
355 F_READ(g_GuiInternal.audio.file, (ft_uint8_t*) &u8, g_GuiInternal.audio.channels, bytesread);
356 sample = u8[0];
357 } else {
358 sample = 0;
359 }
360 if (++g_GuiInternal.audio.atsample == g_GuiInternal.audio.nsamples) {
361 if (g_GuiInternal.audio.mode == PLAY_LOOP) {
362 //cmd_FSeek(audio.f, audio.dataOfs);
363 F_SEEK(g_GuiInternal.audio.file, g_GuiInternal.audio.dataOfs);
364 g_GuiInternal.audio.adpcmPos = 0;
365 g_GuiInternal.audio.adpcmNibble = 0;
366 g_GuiInternal.audio.atsample = 0;
367 } else {
368 //cmd_FClose(audio.f);
369 F_CLOSE(g_GuiInternal.audio.file);
370 clear_bit(g_GuiInternal.audio.flags, FT_AUDIO_FLAGS_BIT_PLAYING);
371 }
372 }
373 }
374 Ft_Gpu_Hal_Wr8(g_GuiInternal.pdirect, SCRATCH_AUDIO_BUF + g_GuiInternal.audio.wp, sample);
375 g_GuiInternal.audio.wp = (g_GuiInternal.audio.wp + 1) & (PLAYBACK_BUFFER_SIZE - 1);
376 fullness++;
377 }
378}

539int adpcm(void)
540{
541 UINT bytesread;
542
543 /* Intel ADPCM step variation table */
544 static const int indexTable[16] = { -1, -1, -1, -1, 2, 4, 6, 8, -1, -1, -1,
545 -1, 2, 4, 6, 8, };
546
547 static const int stepsizeTable[89] = { 7, 8, 9, 10, 11, 12, 13, 14, 16, 17,
548 19, 21, 23, 25, 28, 31, 34, 37, 41, 45, 50, 55, 60, 66, 73, 80, 88,
549 97, 107, 118, 130, 143, 157, 173, 190, 209, 230, 253, 279, 307, 337,
550 371, 408, 449, 494, 544, 598, 658, 724, 796, 876, 963, 1060, 1166,
551 1282, 1411, 1552, 1707, 1878, 2066, 2272, 2499, 2749, 3024, 3327,
552 3660, 4026, 4428, 4871, 5358, 5894, 6484, 7132, 7845, 8630, 9493,
553 10442, 11487, 12635, 13899, 15289, 16818, 18500, 20350, 22385,
554 24623, 27086, 29794, 32767 };
555
556 if (!(g_GuiInternal.audio.adpcmNibble & 0x100)) {
557 if ((g_GuiInternal.audio.adpcmPos % g_GuiInternal.audio.nBlockAlign) == 0) {
558 struct {
559 ft_int16_t predictor;
560 ft_uint8_t index;
561 ft_uint8_t _dummy;
562 } adpcmHdr;
563 F_READ(g_GuiInternal.audio.file, (ft_uint8_t*) &adpcmHdr, sizeof(adpcmHdr), bytesread);
564 g_GuiInternal.audio.adpcmPos += bytesread;
565 g_GuiInternal.audio.adpcmIndex = adpcmHdr.index;
566 g_GuiInternal.audio.adpcmPred = adpcmHdr.predictor;
567 return g_GuiInternal.audio.adpcmPred;
568 }
569
570 g_GuiInternal.audio.adpcmNibble = f_getc(g_GuiInternal.audio.file) | 0x1000;
571 g_GuiInternal.audio.adpcmPos++;
572 }
573
574 int delta = g_GuiInternal.audio.adpcmNibble & 0xf;
575 g_GuiInternal.audio.adpcmNibble >>= 4;
576
577 int step = stepsizeTable[g_GuiInternal.audio.adpcmIndex];
578
579 /* Step 1 - get the delta value and compute next index */
580
581 /* Step 2 - Find new index value (for later) */
582 g_GuiInternal.audio.adpcmIndex += indexTable[delta];
583 if (g_GuiInternal.audio.adpcmIndex < 0)
584 g_GuiInternal.audio.adpcmIndex = 0;
585 if (g_GuiInternal.audio.adpcmIndex > 88)
586 g_GuiInternal.audio.adpcmIndex = 88;
587
588 /* Step 3 - Separate sign and magnitude */
589 int sign = delta & 8;
590 delta = delta & 7;
591
592 /* Step 4 - Compute difference and new predicted value */
593 /*
594 ** Computes 'vpdiff = (delta+0.5)*step/4', but see comment
595 ** in adpcm_coder.
596 */
597 int vpdiff = step >> 3;
598 if (delta & 4)
599 vpdiff += step;
600 if (delta & 2)
601 vpdiff += step >> 1;
602 if (delta & 1)
603 vpdiff += step >> 2;
604
605 if (sign)
606 g_GuiInternal.audio.adpcmPred -= vpdiff;
607 else
608 g_GuiInternal.audio.adpcmPred += vpdiff;
609
610 /* Step 5 - clamp output value */
611 if (g_GuiInternal.audio.adpcmPred > 32767)
612 g_GuiInternal.audio.adpcmPred = 32767;
613 else if (g_GuiInternal.audio.adpcmPred < -32768)
614 g_GuiInternal.audio.adpcmPred = -32768;
615
616 /* Step 6 - Update step value */
617 step = stepsizeTable[g_GuiInternal.audio.adpcmIndex];
618
619 /* Step 6 - Output value */
620 // iprintf("audio.adpcmPred = %04x\n", audio.adpcmPred);
621 return g_GuiInternal.audio.adpcmPred;
622}

624static int audio_seekto(int seek_samples)
625{
626 if (!(g_GuiInternal.audio.flags & FT_AUDIO_FLAGS_PLAYING) || g_GuiInternal.audio.mode == -1 || seek_samples < 0
627 || seek_samples > g_GuiInternal.audio.nsamples)
628 return -1;
629
630 g_GuiInternal.audio.atsample = 0;
631 g_GuiInternal.audio.adpcmPos = 0;
632 g_GuiInternal.audio.adpcmNibble = 0;

634 F_SEEK(g_GuiInternal.audio.file, g_GuiInternal.audio.dataOfs);
635 if (g_GuiInternal.audio.adpcm) {
636 while (g_GuiInternal.audio.atsample++ < seek_samples) {
637 adpcm();
638 }
639 } else {
640 if (g_GuiInternal.audio.bps == 16) {
641 if (g_GuiInternal.audio.channels == 1) {
642 F_SEEK(g_GuiInternal.audio.file, 2 * seek_samples);
643 } else {
644 F_SEEK(g_GuiInternal.audio.file, 4 * seek_samples);
645 }
646 } else if ((g_GuiInternal.audio.format == LINEAR_SAMPLES) && (g_GuiInternal.audio.bps == 8)) {
647 if (g_GuiInternal.audio.channels == 1) {
648 F_SEEK(g_GuiInternal.audio.file, seek_samples);
649 } else {
650 F_SEEK(g_GuiInternal.audio.file, 2 * seek_samples);
651 }
652 } else if (g_GuiInternal.audio.format == ULAW_SAMPLES) {
653 F_SEEK(g_GuiInternal.audio.file, g_GuiInternal.audio.channels * seek_samples);
654 }
655 g_GuiInternal.audio.atsample += seek_samples;
656 }
657 g_GuiInternal.audio.last_sample = g_GuiInternal.audio.played_samples + g_GuiInternal.audio.nsamples - seek_samples;
658 g_GuiInternal.audio.started_samples = g_GuiInternal.audio.played_samples - seek_samples;
659 g_GuiInternal.audio.prev_rdptr = g_GuiInternal.audio.wp = Ft_Gpu_Hal_Rd32(g_GuiInternal.pdirect,
660 REG_PLAYBACK_READPTR) - SCRATCH_AUDIO_BUF;

662 Ft_Gpu_Hal_Wr32(g_GuiInternal.pdirect, REG_PLAYBACK_FORMAT, g_GuiInternal.audio.format);
663 service_playback();
664 return 0;
665}

699int png_scan(FILE *file, int *has_tRNS, int *palsize)
700#else
701int png_scan(FIL *file, int *has_tRNS, int *palsize)
702#endif
703{
704 ft_uint32_t pos = 8, total = 0;
705 UINT actual;
706
707 *has_tRNS = 0;
708 F_SIZE(file, total);
709 while (pos < total) {
710 struct {
711 ft_uint8_t length[4];
712 ft_uint8_t typecode[4];
713 } chunk;
714 F_SEEK(file, pos);
715 F_READ(file, (ft_uint8_t*) &chunk, sizeof(chunk), actual);
716 if (actual != sizeof(chunk))
717 return 0;
718 ft_uint32_t length = (chunk.length[0] << 24) | (chunk.length[1] << 16)
719 | (chunk.length[2] << 8) | (chunk.length[3] << 0);
720 *has_tRNS |= (memcmp(chunk.typecode, "tRNS", 4) == 0);
721 if (memcmp(chunk.typecode, "PLTE", 4) == 0)
722 *palsize = length / 3;
723 pos += (12 + length);
724 }
725 return 1;
726}

729static ft_int16_t _raw_draft_image(FILE *file, struct Hbitmap *gm)
730#else
731static ft_int16_t _raw_draft_image(FIL *file, struct Hbitmap *gm)
732#endif
733{
734 int res;
735 UINT bytesread;
736
737 if (F_SEEK(file, 0) != 0) {
738 printf("WARN: fail to seek to head!\n");
739 }
740 ft_uint16_t magic = be16(file);
741 switch (magic) {
742 case 0x424d: // .bmp starts with 'B' 'M'
743 {
744 FILEHEADER fh;
745 INFOHEADER ih;
746
747 F_SEEK(file, 0);
748 F_READ(file, (ft_uint8_t*) &fh, sizeof(FILEHEADER), bytesread);
749 F_READ(file, (ft_uint8_t*) &ih, sizeof(INFOHEADER), bytesread);
750
751 gm->width = ih.width;
752 gm->height = ih.height;
753 gm->fmt = RGB565;
754 }
755 break;
756
757 case 0xffd8: // .jpg
758 {
759 ft_uint32_t pos;
760#if defined(MSVC_PLATFORM) || defined(BT8XXEMU_PLATFORM)
761 {
762 int fsz;
763
764 F_SIZE(file, fsz);
765 pos = F_TELL(file);
766 if (pos != 2) {
767 if (F_SEEK(file, 2) != 0) {
768 printf("WARN: fail to seek!\n");
769 }
770 pos = F_TELL(file);
771 }
772 ft_uint16_t marker;
773 while ((marker = be16(file)) != 0xffc0 && pos < fsz) {
774 if ((marker & 0xFF00) == 0xFF00) {
775 ft_uint16_t o = be16(file);
776 pos = F_TELL(file) + o - 2;
777 F_SEEK(file, pos);
778 }
779 else {
780 f_getc(file);
781 }
782 }
783 if (marker != 0xffc0) {
784 return ERROR_INVALIDFILE;
785 }
786 }
787#else
788 while (be16(file) != 0xffc0) {
789 ft_uint16_t o = be16(file);
790 pos = F_TELL(file);
791 F_SEEK(file, pos + o - 2);
792 }
793#endif
794 pos = F_TELL(file);
795 F_SEEK(file, pos + 3);
796 gm->height = be16(file);
797 gm->width = be16(file);
798 gm->fmt = RGB565;
799 }
800 break;
801
802 case 0x8950: // .png
803 {
804 F_SEEK(file, 18);
805 gm->width = be16(file);
806 F_SEEK(file, 22);
807 gm->height = be16(file);
808
809 ft_uint8_t bit_depth = f_getc(file);
810 ft_uint8_t color_type = f_getc(file);
811
812 if (bit_depth != 8) {
813 F_CLOSE(file);
814 iprintf("Invalid png");
815 return ERROR_INVALIDFILE;
816 }
817 switch (color_type) {
818 case 0:
819 gm->fmt = L8;
820 break;
821 case 2:
822 gm->fmt = RGB565;
823 break;
824 case 3: {
825 int has_tRNS, palsize;
826 png_scan(file, &has_tRNS, &palsize);
827 gm->fmt = has_tRNS ? PALETTED4444 : PALETTED565;
828 gm->palsize = (palsize - 1);
829 break;
830 }
831 case 6:
832 gm->fmt = ARGB4;
833 break;
834 default:
835 iprintf("NOT supported png file!\n");
836 F_CLOSE(file);
837 return ERROR_INVALIDFILE;
838 }
839 }
840 break;
841
842 default:
843 F_CLOSE(file);
844 iprintf("NOT a bmp, jpg, or png");
845 return ERROR_INVALIDFILE;
846 }
847 return 0;
848}

885int slot(ft_uint32_t h, int handle_num) {
886 // Find the lowest 0 bit in h
887 ft_uint32_t sel = ~h & (h + 1);
888
889 if ((sel & ((1 << NFILES) - 1)) == 0)
890 return -1;
891
892 int index = handle_num;
893 if (sel & 0xf0) index |= 4;
894 if (sel & 0xcc) index |= 2;
895 if (sel & 0xaa) index |= 1;
896 return index;
897 }
