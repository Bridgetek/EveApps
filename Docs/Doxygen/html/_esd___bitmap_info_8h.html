582 static const uint32_t sizes[PALETTED8 - PALETTED565 + 1] = { 256 * 2, 256 * 2, 256 * 4 };
583 uint32_t size = sizes[bitmapInfo->Format - PALETTED565];
584
585 // Verify if there is a file name or flash address specified, or if the flash address can be found in the flash filesystem
586 int32_t paletteFlashAddr = bitmapInfo->Flash ? Esd_LittleFs_LoadFlashAddress(&bitmapInfo->PaletteFlashAddress, bitmapInfo->PaletteFile, NULL) : FA_INVALID;
587 if (bitmapInfo->Flash ? (paletteFlashAddr == FA_INVALID) : (!bitmapInfo->PaletteFile))
588 {
589#ifdef ESD_BITMAPINFO_DEBUG
590 // eve_printf_debug("Bitmap info palette file name is NULL, cannot auto-load\n");
591#endif
592 return GA_INVALID;
593 }
594
595 // Not loaded, load this bitmap palette
596 bitmapInfo->PaletteGpuHandle = Esd_GpuAlloc_Alloc(Esd_GAlloc, size, bitmapInfo->Persistent ? 0 : GA_GC_FLAG);
597 addr = Esd_GpuAlloc_Get(Esd_GAlloc, bitmapInfo->PaletteGpuHandle);
598 if (addr != GA_INVALID)
599 {
600#ifdef ESD_BITMAPINFO_DEBUG
601 eve_printf_debug("Allocated space for bitmap palette\n");
602#endif
603 // Allocation space OK
604 if (
605#ifdef EVE_FLASH_AVAILABLE
606 bitmapInfo->Flash ? !EVE_CoCmd_flashRead_flush(Esd_Host, addr, paletteFlashAddr, size) :
607#endif
608 !EVE_Util_loadRawFile(Esd_Host, addr, bitmapInfo->PaletteFile))
609 {
610#ifdef ESD_BITMAPINFO_DEBUG
611 eve_printf_debug(bitmapInfo->Flash ? "Failed to load palette from flash\n" : "Failed to load palette from file\n");
612#endif
613 // Failed to load from file
614 Esd_GpuAlloc_Free(Esd_GAlloc, bitmapInfo->PaletteGpuHandle);
615 addr = GA_INVALID;
616 }
617 }
618 else
619 {
620#ifdef ESD_BITMAPINFO_DEBUG
621 eve_printf_debug("Unable to allocate space for bitmap palette\n");
622#endif
623 }
624 }
625 }
626 else // FT_80X_ENABLE
627 {
628 eve_printf_debug("Palette not yet implemented for FT80x\n");
629 addr = GA_INVALID;
630 }
631
632 if (addr != GA_INVALID)
633 {
634 addr = ESD_DL_RAM_G_ADDRESS(addr);
635 }
636
637 return addr;
638}
