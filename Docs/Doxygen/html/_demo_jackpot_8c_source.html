751uint16_t stringPixelWidth(const char8_t* text, uint8_t font) {
752 char8_t tempChar;
753 uint16_t length = 0, index;
754 if (text == NULL) {
755 return 0;
756 }
757
758 if (text[0] == '\0') {
759 return 0;
760 }
761
762 index = 0;
763 tempChar = text[index];
764 while (tempChar != '\0') {
765 length += Gpu_Rom_Font_WH(tempChar, font);
766 tempChar = text[++index];
767 }
768
769 return length;
770}
