172static int extractXfont(int numStreamChars, int numStreamCp, CODEPOINT *cp,
173 char *xfontBuffer, char *xfontOut, char *glyphBuffer, char *glyphOut) {
174 int *xf = (int*)xfontBuffer;
175 int xf_new_number_of_characters = ALIGN(numStreamCp, 128);
176 uint32_t bytes_per_glyph = XF_LAYOUT_WIDTH(xf) * XF_LAYOUT_HEIGHT(xf);
177
178 const int xfontHeaderSize = 40;
179 const int xfontWptrSize = 4 * (xf_new_number_of_characters / 128);
180 const int xfontGptrSize = 4 * (xf_new_number_of_characters / 128);
181 const int xfontWidthSize = numStreamCp;
182 int xf_size = xfontHeaderSize + xfontWptrSize + xfontGptrSize + xfontWidthSize;
183 int xf_start_of_Graphic_data = ALIGN(xf_size, 1024);
184
185 // Empty newXfont
186 buff2File(xfontOut, "", 0, fl_write);
187 buff2File(glyphOut, "", 0, fl_write);
188
189 APP_DBG("write newXfont static header");
190 buff2File(xfontOut, xfontBuffer, 4, fl_append);
191 buff2File(xfontOut, (char*)&xf_size, 4, fl_append);
192 buff2File(xfontOut, &xfontBuffer[8], 4 * 6, fl_append);
193 buff2File(xfontOut, (char*)&xf_start_of_Graphic_data, 4, fl_append);
194 buff2File(xfontOut, (char*)&xf_new_number_of_characters, 4, fl_append);
195
196 APP_DBG("write newXfont gptr");
197 for (int i = 0; i < (xf_new_number_of_characters / 128); i++) {
198 unsigned int offset = i * 128 * bytes_per_glyph;
199 buff2File(xfontOut, (char*)&offset, 4, fl_append);
200 }
201
202 APP_DBG("write newXfont wptr");
203 for (int i = 0; i < (xf_new_number_of_characters / 128); i++) {
204 unsigned int offset = i * 128 + 40
205 + 4 * (xf_new_number_of_characters / 128)
206 + 4 * (xf_new_number_of_characters / 128);
207 buff2File(xfontOut, (char*)&offset, 4, fl_append);
208 }
209
210 APP_DBG("Creating new glyph");
211 for (int i = 0; i < numStreamCp; i++) {
212 extractGlyph(cp[i].codepoint, xfontBuffer, xfontOut, glyphBuffer,
213 glyphOut);
214 }
215
216 return APP_OK;
217}

253static char ** streamToArray2D(char *stream, int streamSize, char *xfontIn, int lineWidth, int *lineNum) {
254 static char *ret[100];//maximum 100 line
255 char *line;
256 char* xfont = file2NewBuff(xfontIn, 0, 0, 0);
257 unsigned long c;
258
259 int next = 0, last = 0;
260 int linecount = 0;
261 int start = 0;
262 int byteLinecount = 0;
263 int lineWidthCount = 0, wordStop = 0;
264
265 memset(ret, 0, sizeof(ret));
266#if APPDEBUG
267 APP_DBG("dumpMem(stream, streamSize);");
268 dumpMem(stream, streamSize);
269#endif
270 while (next < streamSize) {
271 next = utf8_next(stream, streamSize, next, &c);
272 byteLinecount += next - last;
273 last = next;
274
275 unsigned char charWidth = getWidth(xfont, c);
276 if (c == CP_SPACE) {//space
277 wordStop = lineWidthCount; // lineWidthCount is pointing to last char of a word
278 }
279 lineWidthCount += charWidth;
280
281 if (lineWidthCount >= lineWidth || c == CP_NEWLINE) { // new line
282 line = memget(byteLinecount + 5);
283 memset(line, 0, byteLinecount + 5);
284 memcpy(line, &stream[start], byteLinecount);
285#if APPDEBUG
286 APP_DBG("dumpMem(line, byteLinecount);");
287 dumpMem(line, byteLinecount);
288#endif
289
290 ret[linecount] = line;
291 linecount++;
292
293 start += byteLinecount;
294 byteLinecount = 0;
295 lineWidthCount = 0;
296 }
297 }
298 *lineNum = linecount;
299
300 return ret;
301}
