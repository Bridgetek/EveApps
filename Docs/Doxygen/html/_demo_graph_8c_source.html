148void set(int32_t x0, int16_t y0,
149 int32_t x1, int16_t y1) {
150 int32_t xd = x1 - x0;
151 int16_t yd = y1 - y0;
152 transform_m = yd / (float_t)xd;
153 if (transform_m < m_min)
154 transform_m = m_min;
155 transform_c = y0 - transform_m * x0;
156}
157void sset(int32_t x0, int16_t y0)
158{
159 transform_c = (float_t)y0 - transform_m * x0;
160}
161int16_t m2s(int32_t x)
162{
163 return (int16_t)(transform_m * x + transform_c);
164}
165int32_t s2m(int16_t y)
166{
167 return (int32_t)(y - transform_c) / transform_m;
168}

174int16_t rsin(int16_t r, uint16_t th)
175{
176 int32_t th4; uint16_t s; int16_t p;
177 th >>= 6; // angle 0-123
178
179 th4 = th & 511;
180 if (th4 & 256)
181 th4 = 512 - th4; // 256->256 257->255, etc
182
183 s = pgm_read_word(sintab_x + th4);
184
185 p = ((uint32_t)s * r) >> 16;
186 if (th & 512)
187 p = -p;
188 return p;
189}
