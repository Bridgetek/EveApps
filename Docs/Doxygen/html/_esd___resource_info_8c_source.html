44ESD_CORE_EXPORT uint32_t Esd_LoadResourceEx(Esd_ResourceInfo *resourceInfo, uint8_t *metadata, uint32_t *imageFormat)
45{
46 EVE_HalContext *phost = Esd_GetHost();
47 int32_t flashAddr = FA_INVALID;
48 uint32_t addr;
49 bool loaded;
50 (void)phost;
51
52 if (metadata)
53 metadata[ESD_METADATA_SIGNATURE] = '\0';
54
55 if (!resourceInfo)
56 {
57 esd_resourceinfo_printf("Resource info is NULL\n");
58 return GA_INVALID;
59 }
60
61 if (!resourceInfo->Compressed && resourceInfo->Type == ESD_RESOURCE_DIRECTFLASH)
62 {
63#ifdef EVE_FLASH_AVAILABLE
64 // Just get the flash address if that's what we want
65 // Calling function may need to ensure that it's not getting a flash address when it's not supported by the resource
66 flashAddr = Esd_LittleFs_LoadFlashAddress(&resourceInfo->FlashAddress, resourceInfo->File, metadata);
67 Esd_LoadResourceMetadata(resourceInfo, metadata);
68 if (!resourceInfo->Compressed)
69 {
70 // If no metadata, or the metadata still specifies uncompressed data, return as direct flash address
71 if (flashAddr != FA_INVALID)
72 return ESD_DL_FLASH_ADDRESS(flashAddr);
73 return GA_INVALID;
74 }
75#else
76 esd_resourceinfo_printf("Flash storage not supported on this platform\n");
77 return GA_INVALID;
78#endif
79 }
80
81 // Get address of specified handle
82 addr = Esd_GpuAlloc_Get(Esd_GAlloc, resourceInfo->GpuHandle);
83 if (addr != GA_INVALID)
84 {
85 return ESD_DL_RAM_G_ADDRESS(addr);
86 }
87
88 // Validate source
89 switch (resourceInfo->Type)
90 {
91 case ESD_RESOURCE_FILE:
92 if (!resourceInfo->File)
93 {
94 esd_resourceinfo_printf("Resource file name is NULL\n");
95 return GA_INVALID;
96 }
97 break;
98 case ESD_RESOURCE_PROGMEM:
99 if (!resourceInfo->ProgMem)
100 {
101 esd_resourceinfo_printf("Resource program memory address is NULL\n");
102 return GA_INVALID;
103 }
104 break;
105 case ESD_RESOURCE_FLASH:
106 case ESD_RESOURCE_DIRECTFLASH:
107 if ((flashAddr = Esd_LittleFs_LoadFlashAddress(&resourceInfo->FlashAddress, resourceInfo->File, metadata)) == FA_INVALID)
108 {
109 esd_resourceinfo_printf("Resource flash address is invalid\n");
110 return GA_INVALID;
111 }
112 break;
113 }
114
115 // Load metadata from file on SD card
116 if (metadata && resourceInfo->Type == ESD_RESOURCE_FILE)
117 {
118 // Generate metafile name
119 size_t nameLen = strlen(resourceInfo->File);
120 if (nameLen + 6 > 128) // ".esdm" NUL
121 return GA_INVALID;
122 char metaFile[128];
123 strcpy_s(metaFile, sizeof(metaFile), resourceInfo->File);
124 strcpy_s(&metaFile[nameLen], sizeof(metaFile) - nameLen, ".esdm");
125
126 // Try to load
127 EVE_Util_readFile(phost, metadata, ESD_METADATA_MAX, metaFile);
128 }
129
130 // Load metadata
131 Esd_LoadResourceMetadata(resourceInfo, metadata);
132
133 // Allocate gpu memory
134 resourceInfo->GpuHandle = Esd_GpuAlloc_Alloc(Esd_GAlloc, resourceInfo->RawSize,
135 (resourceInfo->Persistent ? 0 : GA_GC_FLAG)
136 | ((!resourceInfo->Compressed && ESD_RESOURCE_IS_FLASH(resourceInfo->Type)) ? GA_LOW_FLAG : 0));
137 addr = Esd_GpuAlloc_Get(Esd_GAlloc, resourceInfo->GpuHandle);
138 if (addr == GA_INVALID)
139 {
140 return GA_INVALID;
141 }
142
143 // Attempt to load
144 loaded = false;
145 switch (resourceInfo->Type)
146 {
147 case ESD_RESOURCE_FILE: {
148 switch (resourceInfo->Compressed)
149 {
150 case ESD_RESOURCE_RAW:
151 loaded = EVE_Util_loadRawFile(phost, addr, resourceInfo->File);
152 break;
153 case ESD_RESOURCE_DEFLATE:
154 loaded = EVE_Util_loadInflateFile(phost, addr, resourceInfo->File);
155 break;
156 case ESD_RESOURCE_IMAGE:
157 loaded = EVE_Util_loadImageFile(phost, addr, resourceInfo->File, imageFormat);
158 break;
159 }
160 break;
161 }
162 case ESD_RESOURCE_PROGMEM: {
163 switch (resourceInfo->Compressed)
164 {
165 case ESD_RESOURCE_RAW:
166 EVE_Hal_wrProgMem(phost, addr, resourceInfo->ProgMem, resourceInfo->StorageSize << 2);
167 loaded = true;
168 break;
169 case ESD_RESOURCE_DEFLATE:
170 loaded = EVE_CoCmd_inflate_progMem(phost, addr, resourceInfo->ProgMem, resourceInfo->StorageSize << 2);
171 break;
172 case ESD_RESOURCE_IMAGE:
173 loaded = EVE_CoCmd_loadImage_progMem(phost, addr, resourceInfo->ProgMem, resourceInfo->StorageSize << 2, imageFormat);
174 break;
175 }
176 break;
177 }
178#ifdef EVE_FLASH_AVAILABLE
179 case ESD_RESOURCE_FLASH:
180 case ESD_RESOURCE_DIRECTFLASH: {
181 switch (resourceInfo->Compressed)
182 {
183 case ESD_RESOURCE_RAW:
184 EVE_CoCmd_flashRead(phost, addr, flashAddr, resourceInfo->StorageSize << 2);
185 loaded = EVE_Cmd_waitFlush(phost);
186 break;
187 case ESD_RESOURCE_DEFLATE:
188 loaded = EVE_CoCmd_inflate_flash(phost, addr, flashAddr);
189 break;
190 case ESD_RESOURCE_IMAGE:
191 loaded = EVE_CoCmd_loadImage_flash(phost, addr, flashAddr, imageFormat);
192 break;
193 }
194 break;
195 }
196#endif
197 }
198
199 if (!loaded)
200 {
201 // Failed to load
202 Esd_GpuAlloc_Free(Esd_GAlloc, resourceInfo->GpuHandle);
203 addr = GA_INVALID;
204 }
205
206 // Return address if we found it
207 if (addr != GA_INVALID)
208 {
209 return ESD_DL_RAM_G_ADDRESS(addr);
210 }
211
212 /*
213 // May be implemented by the calling function on GA_INVALID result
214#ifdef EVE_FLASH_AVAILABLE
215 // If flash source, we can use directly from flash (if supported)
216 if (supportDirectFlash && resourceInfo->Type == ESD_RESOURCE_FLASH)
217 {
218 if (flashAddr != FA_INVALID)
219 return ESD_DL_FLASH_ADDRESS(flashAddr);
220 }
221#endif
222 */
223
224 // Fail to load
225 return GA_INVALID;
226}

228ESD_CORE_EXPORT uint32_t Esd_LoadResource(Esd_ResourceInfo *resourceInfo, uint32_t *imageFormat)
229{
230 uint8_t metadata[ESD_METADATA_MAX];
231 return Esd_LoadResourceEx(resourceInfo, metadata, imageFormat);
232}

234ESD_CORE_EXPORT void Esd_FreeResource(Esd_ResourceInfo *resourceInfo)
235{
236 if (!resourceInfo)
237 return;
238
239 Esd_GpuAlloc_Free(Esd_GAlloc, resourceInfo->GpuHandle);
240 resourceInfo->GpuHandle.Id = MAX_NUM_ALLOCATIONS;
241#ifdef ESD_LITTLEFS_FLASH
242 resourceInfo->FlashAddress = FA_INVALID;
243#endif
244}

246ESD_CORE_EXPORT void Esd_ResourcePersist(Esd_ResourceInfo *resourceInfo)
247{
248 Esd_LoadResource(resourceInfo, false);
249}
