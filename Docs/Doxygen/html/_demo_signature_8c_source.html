113void DemoSignature() {
114 uint16_t w, h, x, y, tag;
115
116 int16_t sw = 2 * s_pHalContext->Width / 3;
117 int16_t sh = sw / 3;
118 int16_t ox = (s_pHalContext->Width - sw) / 2;
119 int16_t oy = (2 * s_pHalContext->Height / 3) - sh;
120 uint16_t a = 0;
121
122 x = s_pHalContext->Width * 0.168;
123 y = s_pHalContext->Height * 0.317;
124 w = s_pHalContext->Width - (2 * x);
125 h = s_pHalContext->Height - (2.5 * y);
126
127 InitStar();
128
129 EVE_CoCmd_dlStart(s_pHalContext); // start
130
131 /* Initialize the Star array */
132 EVE_Cmd_wr32(s_pHalContext, BITMAP_HANDLE(BACKGROUND_STAR_HANDLE)); // handle for background stars
133 EVE_Cmd_wr32(s_pHalContext, BITMAP_SOURCE(START_ICON_ADDR)); // Starting address in gram
134 EVE_Cmd_wr32(s_pHalContext, BITMAP_LAYOUT(L4, 16, 32)); // format
135 EVE_Cmd_wr32(s_pHalContext, BITMAP_SIZE(NEAREST, REPEAT, REPEAT, 512, 512));
136
137 EVE_Cmd_wr32(s_pHalContext, CLEAR(1, 1, 1));
138 EVE_CoCmd_memZero(s_pHalContext, RAM_G, 480L * 272L); // Clear the gram frm 1024
139#if defined FT801_ENABLE
140 EVE_CoCmd_cSketch(s_pHalContext, x, y, w, h, RAM_G, L8, 1500L);
141#else
142 EVE_CoCmd_sketch(s_pHalContext, x, y, w, h, RAM_G, L8);
143#endif
144 EVE_Cmd_wr32(s_pHalContext, BITMAP_HANDLE(1)); // handle for background stars
145 EVE_Cmd_wr32(s_pHalContext, BITMAP_SOURCE(RAM_G));
146 EVE_Cmd_wr32(s_pHalContext, BITMAP_LAYOUT(L8, w, h));
147#if defined(FT81X_ENABLE)
148 EVE_Cmd_wr32(s_pHalContext, BITMAP_LAYOUT_H(w >> 10, h >> 9));
149#endif
150 EVE_Cmd_wr32(s_pHalContext, BITMAP_SIZE(NEAREST, BORDER, BORDER, w, h));
151#if defined(FT81X_ENABLE)
152 EVE_Cmd_wr32(s_pHalContext, BITMAP_SIZE_H(w >> 9, h >> 9));
153#endif
154
155 EVE_CoCmd_swap(s_pHalContext);
156 EVE_Cmd_waitFlush(s_pHalContext);
157
158 while (1)
159 {
160 // read the Tag
161 tag = Gesture_GetTag(s_pHalContext);
162 // Clear the GRAM when the Clear button enter by user
163 if (tag == 'O')
164 {
165 Play_Sound(s_pHalContext, 0x50, 255, 0xc0);
166 EVE_CoCmd_dlStart(s_pHalContext);
167 EVE_CoCmd_memZero(s_pHalContext, RAM_G, 480L * 272L); // Clear the gram frm 1024
168 EVE_Cmd_waitFlush(s_pHalContext);
169 }
170 EVE_CoCmd_dlStart(s_pHalContext); // start
171 EVE_Cmd_wr32(s_pHalContext, CLEAR(1, 1, 1));
172 EVE_Cmd_wr32(s_pHalContext, BEGIN(BITMAPS));
173 EVE_Cmd_wr32(s_pHalContext, SAVE_CONTEXT());
174 EVE_Cmd_wr32(s_pHalContext, BLEND_FUNC(SRC_ALPHA, ONE));
175 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(78, 0, 0));
176 // Background animation
177 EVE_Cmd_wr32(s_pHalContext, CMD_LOADIDENTITY);
178 rotate_around(ox, oy, 47 * a);
179 EVE_Cmd_wr32(s_pHalContext, VERTEX2II(0, 0, BACKGROUND_STAR_HANDLE, 1));
180
181 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(0, 40, 0));
182 rotate_around(ox + sw, oy, 53 * a);
183 EVE_Cmd_wr32(s_pHalContext, VERTEX2II(0, 0, BACKGROUND_STAR_HANDLE, 1));
184
185 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(0, 0, 78));
186 rotate_around(ox, oy + sh, 57 * a);
187 EVE_Cmd_wr32(s_pHalContext, VERTEX2II(0, 0, BACKGROUND_STAR_HANDLE, 1));
188 EVE_Cmd_wr32(s_pHalContext, RESTORE_CONTEXT());
189 // Signature area
190 EVE_CoCmd_fgColor(s_pHalContext, 0xffffff); // Set the fg color
191 EVE_Cmd_wr32(s_pHalContext, TAG_MASK(1));
192 EVE_Cmd_wr32(s_pHalContext, TAG('S'));
193 EVE_CoCmd_button(s_pHalContext, x, y, w, h, 31, OPT_FLAT, "");
194
195 // Sketch on the screen
196
197 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(0, 0, 0));
198 EVE_Cmd_wr32(s_pHalContext, BEGIN(BITMAPS));
199 EVE_Cmd_wr32(s_pHalContext, VERTEX2II(x, y, 1, 0));
200
201 if (EVE_Hal_rd8(s_pHalContext, REG_TOUCH_TAG) == 'O')
202 EVE_CoCmd_fgColor(s_pHalContext, 0x003300);
203 else
204 EVE_CoCmd_fgColor(s_pHalContext, 0x005500);
205 EVE_Cmd_wr32(s_pHalContext, COLOR_RGB(255, 255, 255));
206 EVE_Cmd_wr32(s_pHalContext, TAG('O'));
207#if defined(DISPLAY_RESOLUTION_HVGA_PORTRAIT)
208 EVE_CoCmd_button(s_pHalContext, s_pHalContext->Width / 2 - sw / 4, s_pHalContext->Height - sh / 2 - 3 - 30, sw / 2, sh / 2, 28, 0, "CLEAR");
209#else
210 EVE_CoCmd_button(s_pHalContext, s_pHalContext->Width / 2 - sw / 4, s_pHalContext->Height - sh / 2 - 3, sw / 2, sh / 2, 28, 0, "CLEAR");
211#endif
212 EVE_Cmd_wr32(s_pHalContext, TAG_MASK(0));
213 EVE_Cmd_wr32(s_pHalContext, DISPLAY());
214 EVE_CoCmd_swap(s_pHalContext);
215 EVE_Cmd_waitFlush(s_pHalContext);
216 a += 1;
217 }
